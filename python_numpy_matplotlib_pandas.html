3. Matplotlib

V d v biu hm s $y = x^2$

In[]:
import numpy as np
import matplotlib.pyplot as plt

x = np.linspace(-20, 20, 1000)
y = x * x

plt.plot(x, y)
plt.title(" th y = x^2")
plt.xlabel("x")
plt.ylabel("y")
plt.show()

V d v th SIN v COS

In[]:
x = np.arange(0, 3 * np.pi, 0.1)
y_sin = np.sin(x)
y_cos = np.cos(x)

plt.plot(x, y_sin)
plt.plot(x, y_cos)
plt.xlabel('Trc X')
plt.ylabel('Trc Y')
plt.title('Hm SIN v COS trong khong [0, 3]')
plt.legend(['SIN(x)', 'COS(x)'])
plt.show()
